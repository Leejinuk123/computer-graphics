> # 1. 그래픽 만들어보기.

> ## 소스코드 : 
```processing
size(200,200);
translate(50,50);
fill(243,97,166);
ellipse(25,-15,30,30); //왼쪽귀
ellipse(-25,-15,30,30); //오른쪽귀

fill(255,178,245);
ellipse(0,0,80,60); //얼굴
fill(243,97,220);
ellipse(0,0,30,30); //코
fill(0);
ellipse(5,0,10,10); //콧구멍
ellipse(-5,0,10,10); //콧구멍
fill(255);
ellipse(-20,-15,10,10); //왼쪽 눈
ellipse(20,-15,10,10); //오른쪽 눈
fill(0);
ellipse(-20,-15,5,5); //왼쪽 눈동자
ellipse(20,-15,5,5); //오른쪽 눈동자
```
